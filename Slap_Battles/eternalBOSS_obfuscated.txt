btnFloor.MouseButton1Click:Connect(function()
	if btnFloor.BackgroundColor3 == Color3.fromRGB(227, 227, 227) then
		btnFloor.BackgroundColor3 = Color3.fromRGB(147, 254, 115)
		local character = game.Players.LocalPlayer.Character
		local head = character and character:FindFirstChild("Head")

		if head then
			local modelBlock = Instance.new("Part")
			modelBlock.Size = Vector3.new(999999, 0.1, 999999)
			modelBlock.Position =  Vector3.new(4, -4, 15) 
			modelBlock.Anchored = true
			modelBlock.CanCollide = true
			modelBlock.Transparency = 0.7
			modelBlock.Parent = workspace
		end
		btnFloor.Text = "floor created!"
		wait(2.5)
		btnFloor.BackgroundColor3 = Color3.fromRGB(227, 227, 227)
		btnFloor.Text = "Create Floor!"

	end

end)
